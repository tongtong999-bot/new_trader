```python
# RiskManager.calc_sl
def calc_sl(self, price: float, atr: float, signal_type: SignalType) -> float:
    if signal_type == SignalType.LONG:
        return price - (self.cfg.TRAILING_STOP_DISTANCE * atr)
    else:
        return price + (self.cfg.TRAILING_STOP_DISTANCE * atr)

# RiskManager.calc_tp
def calc_tp(self, price: float, atr: float, signal_type: SignalType, r: float = None) -> float:
    if r is None:
        r = self.cfg.FULL_TP_R_MULTIPLE
    if signal_type == SignalType.LONG:
        return price + (r * self.cfg.TRAILING_STOP_DISTANCE * atr)
    else:
        return price - (r * self.cfg.TRAILING_STOP_DISTANCE * atr)
```
